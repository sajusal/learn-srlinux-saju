## BGP Configuration

```srl
A:srl-a# info network-instance default protocols bgp
    network-instance default {
        protocols {
            bgp {
                admin-state enable
                autonomous-system 65000
                router-id 10.10.10.1
                afi-safi evpn {
                    admin-state enable
                }
                afi-safi ipv4-unicast {
                    admin-state enable
                }
                group eBGP-underlay {
                    admin-state enable
                    export-policy export-all
                    import-policy import-all
                    next-hop-self true
                }
                group iBGP-evpn {
                    admin-state enable
                    export-policy export-all
                    import-policy import-all
                    peer-as 65000
                    route-reflector {
                        client true
                        cluster-id 1.1.1.1
                    }
                }
                neighbor 10.1.1.2 {
                    admin-state enable
                    peer-group iBGP-evpn
                }
                neighbor 192.168.10.2 {
                    admin-state enable
                    peer-as 65003
                    peer-group eBGP-underlay
                }
            }
        }
    }
```
